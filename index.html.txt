        function setRequirementType(type, buttonElement) {
            const buttons = document.querySelectorAll('.tab-btn');
            buttons.forEach(btn => btn.classList.remove('active'));
            buttonElement.classList.add('active');


            document.getElementById('reqTypeInput').value = type;
            document.getElementById('emailSubject').value = 'New Real Estate Requirement: ' + type;
        }


        async function handleFormSubmit(event) {
            event.preventDefault();
            const form = event.target;
            const thankYouBox = document.getElementById('thankYouBox');
            const submitBtn = form.querySelector('.btn-submit');


            submitBtn.innerText = "Submitting...";
            submitBtn.disabled = true;


            const formData = new FormData(form);


            try {
                const response = await fetch(form.action, {
                    method: form.method,
                    body: formData,
                    headers: {
                        'Accept': 'application/json'
                    }
                });


                if (response.ok) {
                    form.style.display = 'none';
                    thankYouBox.style.display = 'block';
                } else {
                    alert("Form submit karne me dikkat aayi. Kripya firse try karein ya direct phone karein.");
                    submitBtn.innerText = "Submit Requirement";
                    submitBtn.disabled = false;
                }
            } catch (error) {
                form.style.display = 'none';
                thankYouBox.style.display = 'block';
            }
        }
    </script>
